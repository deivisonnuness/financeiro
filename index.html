🔗 Integração Google Sheets Desconectado
ℹ️ Configure um Google Apps Script para sincronizar seus dados com o Google Sheets. Todos os lançamentos serão exportados automaticamente.
📝 Como Configurar:
1. Abra o Google Sheets e crie uma planilha nova.
2. Vá em Extensões → Apps Script.
3. Cole o código Apps Script fornecido abaixo e clique em Implantar → Novo Implantação → App da Web.
4. Defina "Executar como: Eu" e "Quem tem acesso: Qualquer pessoa".
5. Copie a URL gerada e cole no campo acima.
Ver código Apps Script 
// FinanceFlow — Google Apps Script v2
// Cole este código no Apps Script e reimplante como App da Web:
//   Executar como: Eu  |  Quem acessa: Qualquer pessoa

function doGet(e) {
  var action = (e && e.parameter && e.parameter.action) ? e.parameter.action : 'ping';
  var output;

  if (action === 'ping') {
    output = {success: true, msg: 'FinanceFlow API OK'};

  } else if (action === 'export') {
    var sheetId = e.parameter.sheetId || '';
    var ss = sheetId ? SpreadsheetApp.openById(sheetId) : SpreadsheetApp.getActiveSpreadsheet();
    output = {success: true, data: exportData(ss)};

  } else {
    output = {success: false, msg: 'Acao desconhecida: ' + action};
  }

  return ContentService
    .createTextOutput(JSON.stringify(output))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    var raw = e.postData ? e.postData.contents : '{}';
    var data = JSON.parse(raw);
    var sheetId = data.sheetId || '';
    var ss = sheetId
      ? SpreadsheetApp.openById(sheetId)
      : SpreadsheetApp.getActiveSpreadsheet();

    if (data.action === 'sync') {
      syncData(ss, data);
      return ContentService
        .createTextOutput(JSON.stringify({success: true, msg: 'Sincronizado com sucesso!'}))
        .setMimeType(ContentService.MimeType.JSON);
    }

    return ContentService
      .createTextOutput(JSON.stringify({success: false, msg: 'Acao invalida'}))
      .setMimeType(ContentService.MimeType.JSON);

  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({success: false, msg: err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function syncData(ss, data) {
  if (data.receitas !== undefined) {
    var rs = getOrCreateSheet(ss, 'Receitas');
    rs.clearContents();
    rs.getRange(1,1,1,6).setValues([['Nome','Valor','Data','Categoria','Recorrente','ID']]);
    if (data.receitas.length > 0) {
      rs.getRange(2,1,data.receitas.length,6).setValues(
        data.receitas.map(function(r){ return [r.nome,r.valor,r.data||'',r.categoria||'',r.recorrente||'',r.id||'']; })
      );
    }
  }

  if (data.despesas !== undefined) {
    var ds = getOrCreateSheet(ss, 'Despesas');
    ds.clearContents();
    ds.getRange(1,1,1,9).setValues([['Descricao','Valor','Data','Categoria','Tipo','Forma','Cartao','Pago','ID']]);
    if (data.despesas.length > 0) {
      ds.getRange(2,1,data.despesas.length,9).setValues(
        data.despesas.map(function(d){ return [d.nome,d.valor,d.data||'',d.categoria||'',d.tipo||'',d.forma||'',d.cartao||'',d.pago?'Sim':'Nao',d.id||'']; })
      );
    }
  }

  if (data.investimentos !== undefined) {
    var is = getOrCreateSheet(ss, 'Investimentos');
    is.clearContents();
    is.getRange(1,1,1,5).setValues([['Nome','Valor','Tipo','Data','ID']]);
    if (data.investimentos.length > 0) {
      is.getRange(2,1,data.investimentos.length,5).setValues(
        data.investimentos.map(function(i){ return [i.nome,i.valor,i.tipo||'',i.data||'',i.id||'']; })
      );
    }
  }
}

function exportData(ss) {
  var result = {receitas:[], despesas:[], investimentos:[]};
  try {
    var rs = ss.getSheetByName('Receitas');
    if (rs && rs.getLastRow() > 1) {
      result.receitas = rs.getRange(2,1,rs.getLastRow()-1,6).getValues()
        .map(function(r){ return {nome:r[0],valor:Number(r[1]),data:r[2],categoria:r[3],recorrente:r[4],id:r[5]}; });
    }
  } catch(e){}
  try {
    var ds = ss.getSheetByName('Despesas');
    if (ds && ds.getLastRow() > 1) {
      result.despesas = ds.getRange(2,1,ds.getLastRow()-1,9).getValues()
        .map(function(d){ return {nome:d[0],valor:Number(d[1]),data:d[2],categoria:d[3],tipo:d[4],forma:d[5],cartao:d[6],pago:d[7]==='Sim',id:d[8]}; });
    }
  } catch(e){}
  try {
    var is = ss.getSheetByName('Investimentos');
    if (is && is.getLastRow() > 1) {
      result.investimentos = is.getRange(2,1,is.getLastRow()-1,5).getValues()
        .map(function(i){ return {nome:i[0],valor:Number(i[1]),tipo:i[2],data:i[3],id:i[4]}; });
    }
  } catch(e){}
  return result;
}

function getOrCreateSheet(ss, name) {
  return ss.getSheetByName(name) || ss.insertSheet(name);
}
